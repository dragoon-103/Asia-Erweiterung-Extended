@never_primary_score = -1000

#### IMPORTANT NOTE: We use the set_color_from_title effect, which adds a slight random variation to the color.
#### This means that if any color component is too close to 0 or 255, it may wrap around and turn into something
#### completely different. E.g. (225, 175, 0) would wrap to (~225, ~175, 255!!) which is purple (not gold).
#### From experimenting, a 5 unit buffer isn't quite enough, so we use 10. All colors are [10, 245].

# Used for Han + Ming
e_color_yellow = {
	color = { 245 245 10 }
	landless = yes
	
	no_automatic_claims = yes
	de_jure_drift_disabled = yes

	can_create = { always = no }
	can_create_on_partition = { always = no}

	capital = c_luoyang

	ai_primary_priority = {
		add = @never_primary_score
	}
}
