# China united by involved Chinese character
mandate_of_heaven_chinese_ending_decision = {
	decision_group_type = major
	title = mandate_of_heaven_chinese_ending_decision
	desc = mandate_of_heaven_chinese_ending_decision_desc
	selection_tooltip = mandate_of_heaven_chinese_ending_decision_tooltip

	is_invisible = yes
	sort_order = 80
	
	# TODO Change
	picture = {
		reference = "gfx/interface/illustrations/decisions/fp2_decision_struggle_hostility.dds"
	}
	extra_picture = "gfx/interface/illustrations/struggle_decision_buttons/fp2_decision_hostility.dds"

	is_shown = {
		is_landless_adventurer = no
		any_character_struggle = {
			involvement = involved
			is_struggle_type = mandate_of_heaven_struggle
		}
	}

	is_valid = {
		# Independent, involved (chinese culture), and not currently Emperor
		is_independent_ruler = yes
		custom_tooltip = {
			text = mandate_of_heaven_already_emperor_tt
			NOT = { is_chinese_emperor_trigger = yes }
		}

		any_character_struggle = {
			involvement = involved
			is_struggle_type = mandate_of_heaven_struggle
			is_struggle_phase = struggle_mandate_of_heaven_phase_precious_harmony
		}
		
		# No other involved/interloper ruler owns an empire title
		NOT = {
			any_independent_ruler = {
				NOT = { this = root }
				primary_title = { tier = tier_empire }
				any_character_struggle = {
					is_struggle_type = mandate_of_heaven_struggle
				}
			}
		}

		# Holds at least one kingdom
		primary_title = {
			tier >= tier_kingdom
		}

		# Holds enough counties
		controls_mandate_of_heaven_counties_trigger = yes
		
		# Legitimate and famous enough
		legitimacy_level >= 4
		prestige_level >= high_prestige_level

		# Positive vassal opinion
		any_powerful_vassal = {
			count = all
			opinion = {
				target = root
				value >= 0
			}
		}
	}

	effect = {
		if = {
			limit = { has_dlc_feature = legends }
			legend_seed_struggle_ending_effect = {
				ENDER = root
				STRUGGLE = mandate_of_heaven_struggle
			}
		}
		
		##### Major Effects #####
		add_prestige = 20000
		dynasty = {
			add_dynasty_prestige = 10000

			add_dynasty_modifier = {
				modifier = aep_legitimate_dynasty
				years = 100
			}
		}

		add_character_modifier = {
			modifier = aep_china_reunifier
		}

		custom_tooltip = {
			text = mandate_of_heaven_struggle_form_china_title_tt
			trigger_event = aep_mandate_struggle.0200
		}
	}

	cost = {}

	ai_check_interval = 120

	ai_potential = { always = yes }

	ai_will_do = { base = 100 }
}
